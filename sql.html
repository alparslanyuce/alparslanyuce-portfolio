CREATE FUNCTION dbo.EBFNC_SIPFORECAST1  
(  
    @STOK_KODU VARCHAR(50),  
    @TARIH DATE,  
    @MIKTAR DECIMAL(18,2)  
)  
RETURNS INT  
AS  
BEGIN  

    DECLARE @SONUC INT = 0  
    DECLARE @AY INT  
    DECLARE @YIL INT  

    DECLARE @FORECAST_MIKTAR DECIMAL(18,2) = 0  
    DECLARE @SIP_BAKIYE DECIMAL(18,2) = 0  

    SET @AY = MONTH(@TARIH)  
    SET @YIL = YEAR(@TARIH)  

    ;WITH FORECAST AS  
    (  
        SELECT   
            t.STOK_KODU,  
            v.AyNumarasi,  
            CASE   
                WHEN v.AyNumarasi < MONTH(GETDATE())   
                    THEN YEAR(GETDATE()) + 1  
                ELSE YEAR(GETDATE())  
            END AS YIL,  
            v.Miktar  
        FROM TBLSIP12TMP t  
        CROSS APPLY  
        (  
            VALUES  
                (1 , t.MIKTAR_1),  
                (2 , t.MIKTAR_2),  
                (3 , t.MIKTAR_3),  
                (4 , t.MIKTAR_4),  
                (5 , t.MIKTAR_5),  
                (6 , t.MIKTAR_6),  
                (7 , t.MIKTAR_7),  
                (8 , t.MIKTAR_8),  
                (9 , t.MIKTAR_9),  
                (10, t.MIKTAR_10),  
                (11, t.MIKTAR_11),  
                (12, t.MIKTAR_12)  
        ) v(AyNumarasi, Miktar)  
        WHERE t.STOK_KODU = @STOK_KODU  
    )  

    SELECT @FORECAST_MIKTAR = ISNULL(SUM(Miktar),0)
    FROM FORECAST
    WHERE AyNumarasi = @AY
    AND YIL = @YIL;

    IF (@FORECAST_MIKTAR < @MIKTAR)
        RETURN 1;

    RETURN 0;

END

SELECT 
    t.STOK_KODU,
    t.TARIH,
    t.TESTAR,
    t.HABERTAR,
    t.SIPTAR,

    DATEDIFF(DAY, t.SIPTAR, t.TARIH) AS siparis_ihtiyac_fark,
    DATEDIFF(DAY, t.TESTAR, t.TARIH) AS teslim_ihtiyac_fark,
    DATEDIFF(DAY, t.HABERTAR, t.TESTAR) AS bildirim_teslim_fark,

    s.TEMIN_SURESI,
    c.NAKLIYE_SURESI

FROM TBLMRPSIP t
LEFT JOIN TBLSTSABIT s ON s.STOK_KODU = t.STOK_KODU
LEFT JOIN TBLCARISTOK c ON c.CARI_KOD = t.SATICI_KODU;

SELECT
    STOK_KODU,
    DEPO_KODU,

    SUM(CASE WHEN STHAR_GCKOD = 'G' THEN STHAR_GCMIK ELSE 0 END) AS GIRIS_MIKTAR,
    SUM(CASE WHEN STHAR_GCKOD = 'C' THEN STHAR_GCMIK ELSE 0 END) AS CIKIS_MIKTAR,

    SUM(
        CASE 
            WHEN STHAR_GCKOD = 'G' THEN STHAR_GCMIK 
            WHEN STHAR_GCKOD = 'C' THEN -STHAR_GCMIK 
            ELSE 0 
        END
    ) AS NET_MIKTAR

FROM TBLSTHAR
GROUP BY STOK_KODU, DEPO_KODU;
