FORECAST

Demand Forecast Function

ERP aylık tahmin ve sipariş karşılaştırma fonksiyonu

CREATE FUNCTION dbo.EBFNC_SIPFORECAST1 ( @STOK_KODU VARCHAR(50), @TARIH DATE, @MIKTAR DECIMAL(18,2) ) RETURNS INT AS BEGIN DECLARE @SONUC INT = 0 DECLARE @AY INT DECLARE @YIL INT DECLARE @FORECAST_MIKTAR DECIMAL(18,2) = 0 DECLARE @SIP_BAKIYE DECIMAL(18,2) = 0 SET @AY = MONTH(@TARIH) SET @YIL = YEAR(@TARIH) ;WITH FORECAST AS ( SELECT t.STOK_KODU, v.AyNumarasi, v.Miktar FROM TBLSIP12TMP t CROSS APPLY (VALUES (1,t.MIKTAR_1),(2,t.MIKTAR_2),(3,t.MIKTAR_3), (4,t.MIKTAR_4),(5,t.MIKTAR_5),(6,t.MIKTAR_6), (7,t.MIKTAR_7),(8,t.MIKTAR_8),(9,t.MIKTAR_9), (10,t.MIKTAR_10),(11,t.MIKTAR_11),(12,t.MIKTAR_12) ) v(AyNumarasi,Miktar) WHERE t.STOK_KODU = @STOK_KODU ) SELECT @FORECAST_MIKTAR = SUM(Miktar) FROM FORECAST WHERE AyNumarasi = @AY; IF (@FORECAST_MIKTAR < @MIKTAR) SET @SONUC = 1; RETURN @SONUC; END
