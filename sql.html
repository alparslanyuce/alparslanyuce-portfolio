Demand Forecast Function

CREATE FUNCTION dbo.EBFNC_SIPFORECAST1 ( @STOK_KODU VARCHAR(50), @TARIH DATE, @MIKTAR DECIMAL(18,2) ) RETURNS INT AS BEGIN DECLARE @SONUC INT = 0 DECLARE @AY INT DECLARE @YIL INT DECLARE @FORECAST_MIKTAR DECIMAL(18,2) = 0 DECLARE @SIP_BAKIYE DECIMAL(18,2) = 0 SET @AY = MONTH(@TARIH) SET @YIL = YEAR(@TARIH) ;WITH FORECAST AS ( SELECT t.STOK_KODU, v.AyNumarasi, CASE WHEN v.AyNumarasi < MONTH(GETDATE()) THEN YEAR(GETDATE()) + 1 ELSE YEAR(GETDATE()) END AS YIL, v.Miktar FROM TBLSIP12TMP t CROSS APPLY ( VALUES (1 , t.MIKTAR_1), (2 , t.MIKTAR_2), (3 , t.MIKTAR_3), (4 , t.MIKTAR_4), (5 , t.MIKTAR_5), (6 , t.MIKTAR_6), (7 , t.MIKTAR_7), (8 , t.MIKTAR_8), (9 , t.MIKTAR_9), (10, t.MIKTAR_10), (11, t.MIKTAR_11), (12, t.MIKTAR_12) ) v(AyNumarasi, Miktar) WHERE t.STOK_KODU = @STOK_KODU ) SELECT @FORECAST_MIKTAR = ISNULL(SUM(Miktar),0) FROM FORECAST WHERE AyNumarasi = @AY AND YIL = @YIL SELECT @SIP_BAKIYE = ISNULL(SUM(A.STHAR_GCMIK - A.FIRMA_DOVTUT),0) FROM TBLSIPATRA A WITH(NOLOCK) WHERE A.STOK_KODU = @STOK_KODU AND A.STHAR_FTIRSIP = '6' AND A.STHAR_HTUR = 'H' AND MONTH(A.D_YEDEK10) = @AY AND YEAR(A.D_YEDEK10) = @YIL AND (A.STHAR_GCMIK - A.FIRMA_DOVTUT) > 0 IF (@SIP_BAKIYE + @MIKTAR) > @FORECAST_MIKTAR RETURN 1 RETURN 0 END

MRP Lead Time Analysis

SELECT t.STOK_KODU, t.TARIH AS ihtiyac_tarihi, t.TESTAR AS teslim_tarihi, t.HABERTAR AS bildirim_tarihi, t.SIPTAR AS siparis_tarihi, DATEDIFF(DAY, t.SIPTAR, t.TARIH) AS siparis_ihtiyac_fark, DATEDIFF(DAY, t.TESTAR, t.TARIH) AS teslim_ihtiyac_fark, DATEDIFF(DAY, t.HABERTAR, t.TESTAR) AS bildirim_teslim_fark, s.TEMIN_SURESI, c.NAKLIYE_SURESI FROM TBLMRPSIP t LEFT JOIN TBLSTSABIT s ON s.STOK_KODU = t.STOK_KODU AND s.SUBE_KODU = t.ISLETME_KODU LEFT JOIN TBLCARISTOK c ON c.CARI_KOD = t.SATICI_KODU WHERE t.STOK_KODU LIKE 'LF024013000403%' ORDER BY t.TARIH

Stock Movement Analysis

SELECT STOK_KODU, DEPO_KODU, SUM(CASE WHEN STHAR_GCKOD = 'G' THEN STHAR_GCMIK ELSE 0 END) AS GIRIS_MIKTAR, SUM(CASE WHEN STHAR_GCKOD = 'C' THEN STHAR_GCMIK ELSE 0 END) AS CIKIS_MIKTAR, SUM( CASE WHEN STHAR_GCKOD = 'G' THEN STHAR_GCMIK WHEN STHAR_GCKOD = 'C' THEN -STHAR_GCMIK ELSE 0 END ) AS NET_MIKTAR FROM TBLSTHAR WHERE STOK_KODU = 'LF022602088026-Y' AND DEPO_KODU = '101' GROUP BY STOK_KODU, DEPO_KODU;
