🏭 MRP Lead Time Analysis

TR: Tedarik, üretim ve teslim sürelerinin analiz edilmesi

EN: Lead time analysis across supply chain and production stages

SELECT 
    STOK_KODU,
    TARIH,
    DATEDIFF(DAY, SIPTAR, TARIH) AS SIPARIS_FARK
FROM TBLMRPSIP

📦 Inventory Movement (Stock Flow)

TR: Depo bazlı giriş-çıkış ve net stok hesaplama

EN: Warehouse-based inventory inflow/outflow calculation

SELECT
  STOK_KODU,
  SUM(CASE WHEN GCKOD='G' THEN MIKTAR ELSE -MIKTAR END) AS NET
FROM TBLSTHAR
GROUP BY STOK_KODU

🧾 Invoice & Stock Transaction

TR: Fatura ve stok hareketlerinin ERP bazlı analizi

EN: ERP-based invoice and stock transaction tracking

SELECT F.TARIH, S.STOK_KODU, F.FATIRS_NO
FROM TBLFATUIRS F
JOIN TBLSTHAR S ON F.FATIRS_NO = S.FISNO
